import requests

def check_honeypot(token_address: str) -> dict:
    """Check any token - chain is auto-detected!"""
    response = requests.post(
        "https://yelling-patience-nirvagold-0a943e82.koyeb.app/v1/honeypot/check",
        json={"token_address": token_address}
    )
    return response.json()

# Works for ANY chain - no need to specify!
result = check_honeypot("0x833589fCD6eDb6E08f4c7C32D4f71b54bdA02913")  # USDC on Base

if result["success"]:
    data = result["data"]
    print(f"Chain: {data['chain_name']}")  # Auto-detected: "Base"
    print(f"Token: {data['token_symbol']}")
    print(f"Honeypot: {data['is_honeypot']}")
    print(f"Risk: {data['risk_score']}/100")
